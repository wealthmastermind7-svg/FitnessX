## 4. Results Screen Code

### `BodyScanResultsScreen.tsx`

```tsx
import React from "react";
import { View, Text, StyleSheet, ScrollView } from "react-native";
import { LineChart } from "react-native-chart-kit";
import { BodyScan } from "@/types/bodyScan";

interface Props {
  currentScan: BodyScan;
  previousScan?: BodyScan;
  history: BodyScan[];
}

export default function BodyScanResultsScreen({
  currentScan,
  previousScan,
  history,
}: Props) {
  const delta = previousScan
    ? {
        shoulders:
          currentScan.shoulderWidthCm - previousScan.shoulderWidthCm,
        hips: currentScan.hipWidthCm - previousScan.hipWidthCm,
        posture:
          currentScan.postureScore - previousScan.postureScore,
      }
    : null;

  return (
    <ScrollView style={styles.container}>
      {/* HERO SCORE */}
      <View style={styles.hero}>
        <Text style={styles.heroLabel}>POSTURE SCORE</Text>
        <Text style={styles.heroValue}>
          {Math.round(currentScan.postureScore)}
        </Text>
        <Text style={styles.heroSub}>out of 100</Text>
      </View>

      {/* METRICS */}
      <View style={styles.metrics}>
        <Metric
          label="Shoulder Width"
          value={`${currentScan.shoulderWidthCm.toFixed(1)} cm`}
          delta={delta?.shoulders}
        />
        <Metric
          label="Hip Width"
          value={`${currentScan.hipWidthCm.toFixed(1)} cm`}
          delta={delta?.hips}
        />
        <Metric
          label="Waist Ratio"
          value={currentScan.waistRatio.toFixed(3)}
        />
      </View>

      {/* CHART */}
      {history.length > 1 && (
        <ProgressChart history={history} />
      )}

      {/* CTA */}
      <View style={styles.cta}>
        <Text style={styles.ctaText}>
          Track changes weekly for best accuracy
        </Text>
      </View>
    </ScrollView>
  );
}
```

## 5. Metric Card Component

```tsx
function Metric({
  label,
  value,
  delta,
}: {
  label: string;
  value: string;
  delta?: number;
}) {
  const deltaColor =
    delta === undefined
      ? "#AAA"
      : delta >= 0
      ? "#4CAF50"
      : "#FF5252";

  return (
    <View style={styles.metricCard}>
      <Text style={styles.metricLabel}>{label}</Text>
      <Text style={styles.metricValue}>{value}</Text>
      {delta !== undefined && (
        <Text style={{ color: deltaColor }}>
          {delta >= 0 ? "+" : ""}
          {delta.toFixed(1)}
        </Text>
      )}
    </View>
  );
}
```

## 6. Progress Chart (Key Retention Driver)

### Chart Component

```tsx
function ProgressChart({ history }: { history: BodyScan[] }) {
  return (
    <View style={styles.chartContainer}>
      <Text style={styles.chartTitle}>
        Shoulder Width Progress
      </Text>
      <LineChart
        data={{
          labels: history.map((_, i) => `#${i + 1}`),
          datasets: [
            {
              data: history.map(
                (s) => s.shoulderWidthCm
              ),
            },
          ],
        }}
        width={340}
        height={220}
        chartConfig={{
          backgroundColor: "#0A0E1A",
          backgroundGradientFrom: "#0A0E1A",
          backgroundGradientTo: "#0A0E1A",
          decimalPlaces: 1,
          color: () => "#FF6B6B",
          labelColor: () => "#AAA",
        }}
        bezier
        style={{ borderRadius: 16 }}
      />
    </View>
  );
}
```
